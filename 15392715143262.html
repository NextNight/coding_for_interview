import time

def meger(left, right):
'''åˆå¹¶'''
    rs = []
    i, j = 0, 0
    while i < len(left) and j < len(right):
        if left[i] < right[j]:
            rs.append(left[i])
            i += 1
        else:
            rs.append(right[j])
            j += 1
    if i < len(left):
        rs.extend(left[i:])  # rs+=left[i:]
    if j < len(right):
        rs.extend(right[j:])  # rs+=right[i:]
    return rs

def merge_sort(nums):
    '''é€’å½’ï¼š'''
    if len(nums) <= 1:
        return nums
    mid = len(nums) // 2
    # åˆ†å‰²
    left, right = nums[:mid], nums[mid:]
    # æ’åº
    left = merge_sort(left)
    right = merge_sort(right)
    # åˆå¹¶
    return meger(left, right)


def merge_sort_2(nums):
    '''éé€’å½’'''
    nums = [[num] for num in nums ]
    while len(nums)>1:
        ns = []
        for i in range(round(len(nums)/2)):
            ns.append(meger(nums[i*2],nums[i*2+1] if i*2+1<=len(nums)-1 else []))
        nums = ns
    return nums[0]
